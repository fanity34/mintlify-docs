---
title: "Order Details"
api: "GET https://api.centralcart.com.br/v1/app/order/{order_id}"
authMethod: "bearer"
description: "Obtém os detalhes de um pedido."
---

### Response

<ResponseField name="client_identifier" type="string">
  Identificador do cliente
</ResponseField>

<ResponseField name="variables" type="string">
  Variáveis de checkout
</ResponseField>

<ResponseField name="client_email" type="string">
  O email do cliente
</ResponseField>

<ResponseField name="client_name" type="string">
  O o nome do cliente
</ResponseField>

<ResponseField name="client_document" type="string | null">
  O documento do cliente
</ResponseField>

<ResponseField name="client_phone" type="string | null">
  O telefone do cliente
</ResponseField>

<ResponseField name="client_discord" type="string | null">
  O Discord ID do cliente
</ResponseField>

<ResponseField name="price" type="string">
  O valor total do pedido
</ResponseField>

<ResponseField name="net_received" type="string">
  O valor recebido pelo gateway
</ResponseField>

<ResponseField name="internal_id" type="string">
  O identificador do pedido
</ResponseField>

<ResponseField name="external_id" type="string">
  O identificador do pedido pelo gateway
</ResponseField>

<ResponseField name="promo_code" type="string | null">
  Cupom de desconto utilizado na compra
</ResponseField>

<ResponseField name="promo_type" type="string | null">
  Tipo do cupom de desconto: `PERCENTAGE` ou `PRICE`
</ResponseField>

<ResponseField name="promo_value" type="string | null">
  Valor do cupom de desconto
</ResponseField>

<ResponseField name="gateway" type="string">
  Método de pagamento utilizado
</ResponseField>

<ResponseField name="status" type="string">
  Status do pedido: `APPROVED`, `REJECTED`, `CANCELED`, `REFUNDED`, `CHARGEDBACK`
</ResponseField>

<ResponseField name="currency" type="string">
  Moeda utilizada na compra: `BRL`, `USD`, `EUR`
</ResponseField>

<ResponseField name="packages" type="array">
  Pacotes do pedido
</ResponseField>
