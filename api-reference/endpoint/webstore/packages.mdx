---
title: "List Packages"
api: "GET https://api.centralcart.com.br/v1/webstore/package"
description: "Lista pacotes disponíveis."
---

<ParamField header="x-store-domain" type="string" required>
  Domínio da sua loja (ex: sualoja.centralcart.com.br)
</ParamField>

<ParamField query="v" default="2" required>
  Versão da API
</ParamField>

<ParamField query="q" type="string">
  Busca por nome
</ParamField>

<ParamField query="order_by" type="string">
  Ordenação por `id` ou `price` no formato `campo:asc|desc` (padrão: `id:asc`)
</ParamField>

<ParamField query="category_id" type="string | number">
  Filtra por categoria. Use `all` para obter produtos de todas as categoria.
</ParamField>

<ParamField query="id" type="number | number[]">
  Filtra por um ou mais IDs
</ParamField>

<ParamField query="include_variations" type="boolean">
  Inclui variações quando `true` (por padrão retorna apenas o pacote principal)
</ParamField>

<ParamField query="load_parent" type="boolean">
  Inclui dados resumidos do pacote pai
</ParamField>

<ParamField query="limit" type="number">
  Quantidade por página (default interno)
</ParamField>

<ParamField query="page" type="number">
  Página para paginação (default: 1)
</ParamField>
