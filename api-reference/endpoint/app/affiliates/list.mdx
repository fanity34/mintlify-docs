title: "List Affiliate"
api: "GET https://api.centralcart.com.br/v1/app/affiliate"
authMethod: "bearer"
description: "Retorna a lista de afiliados criados."

<ParamField query="page" type="number">
  Número da página para paginação
</ParamField>

<ParamField query="search" type="string">
  Busca por nome ou email do afiliado
</ParamField>

<ParamField query="email" type="string">
  Filtrar por email específico do afiliado
</ParamField>

<ParamField query="all" type="boolean">
  Se `true`, retorna todos os afiliados sem paginação
</ParamField>
