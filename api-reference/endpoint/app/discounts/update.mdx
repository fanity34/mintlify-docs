---
title: "Update Discount"
api: "PATCH https://api.centralcart.com.br/v1/app/discount/{discount-id}"
authMethod: "bearer"
description: "Atualiza um cupom de desconto."
---

<ParamField path="discount-id" type="string">
  O ID do desconto a ser atualizado
</ParamField>

<ParamField body="coupon" type="string">
  O cupom que será atualizado
</ParamField>

<ParamField body="applies_to" type="number[]">
  Lista de ID dos produtos em que o cupom poderá ser aplicado
</ParamField>

<ParamField body="type" type="string">
  O tipo do desconto, podendo ser `PRICE` ou `PERCENTAGE`
</ParamField>

<ParamField body="value" type="number">
  O valor do desconto
</ParamField>

<ParamField body="max_uses" type="number">
  A quantidade máxima de usos que o desconto pode ter
</ParamField>

<ParamField body="expires_in" type="date">
  A data de expiração do desconto
</ParamField>
