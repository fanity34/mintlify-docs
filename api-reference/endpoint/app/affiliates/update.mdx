---

<ParamField path="affiliate-id" type="string" required>
  O ID do afiliado a ser atualizado
</ParamField>

<ParamField body="name" type="string">
  Nome do afiliado (máximo 56 caracteres)
</ParamField>

<ParamField body="ref_code" type="string">
  Código de referência único do afiliado (apenas letras, números, hífen e underscore)
</ParamField>

<ParamField body="percent_commission" type="number">
  Percentual de comissão do afiliado (entre 1 e 100)
</ParamField>

<ParamField body="email" type="string">
  Email do afiliado
</ParamField>

<ParamField body="balance" type="number">
  Saldo do afiliado
</ParamField>

<ParamField body="min_withdraw" type="number">
  Valor mínimo para saque
</ParamField>
