---
title: "List Posts"
api: "GET https://api.centralcart.com.br/v1/webstore/post"
description: "Lista posts recentes do blog da loja."
---

<ParamField header="x-store-domain" type="string" required>
  Domínio da sua loja (ex: sualoja.centralcart.com.br)
</ParamField>

<ParamField query="page" type="number">
  Página para paginação (default: 1)
</ParamField>

### Response

<ResponseField name="meta" type="object">
  Metadados de paginação
</ResponseField>

<ResponseField name="data" type="array">
  Lista de posts
</ResponseField>
