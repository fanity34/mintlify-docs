<ParamField header="x-store-domain" type="string" required>
  Domínio da sua loja (ex: sualoja.centralcart.com.br)
</ParamField>

<ParamField body="gateway" type="string" required>

</ParamField>

<ParamField body="client_email" type="string" required>
  Email do comprador
</ParamField>

<ParamField body="client_name" type="string" required>
  Nome do comprador (nome completo)
</ParamField>

<ParamField body="client_phone" type="string">
  Telefone do comprador
</ParamField>

<ParamField body="client_document" type="string">
  Documento (CPF) do comprador
</ParamField>

<ParamField body="client_discord" type="string">
  Discord ID do comprador
</ParamField>

<ParamField body="terms" type="boolean" required>
  Aceite dos termos
</ParamField>

<ParamField body="fields" type="object">
  Campos de checkout (ex: `client_identifier`)
</ParamField>

<ParamField body="cart" type="array">
  <ParamField body="package_id" type="number" required>
    ID do pacote
  </ParamField>
  <ParamField body="quantity" type="number" required>
    Quantidade
  </ParamField>
  <ParamField body="options" type="object">
    Variáveis por pacote
  </ParamField>
  <ParamField body="fields" type="object">
    Alias para `options`
  </ParamField>
</ParamField>

<ParamField body="coupon" type="string">
  Cupom de desconto
</ParamField>
