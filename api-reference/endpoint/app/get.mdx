title: "Application Details"

authMethod: "bearer"
description: "Obtém informações sobre a sua loja."
---

### Response

<ResponseField name="id" type="number">
  O ID da sua loja
</ResponseField>

<ResponseField name="name" type="string">
  O nome da sua loja
</ResponseField>

<ResponseField name="plan" type="string">
  O plano da sua loja
</ResponseField>

<ResponseField name="overdue_date" type="string">
  A data expiração da sua assinatura
</ResponseField>

<ResponseField name="url" type="string">
  A url da sua loja
</ResponseField>

<ResponseField name="logo" type="string">
  A logo da sua loja
</ResponseField>

<ResponseField name="primary_color" type="string">
  A cor primária da sua loja em hexadecimal
</ResponseField>
